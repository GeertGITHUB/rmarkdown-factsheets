\newcommand{\headertemplate}[2]{
  \thispagestyle{empty}
  \begin{tikzpicture}[overlay, remember picture]
    \draw let \p1 = (current page.west), \p2 = (current page.east) in
      node[minimum width=\x2-\x1, minimum height=2cm, draw=urbanblue, rectangle, fill=urbanblue, anchor=north west, align=left, text width=\x2-\x1] at ($(current page.north west)$) {\huge\bfseries\color{white} \quad #1};
    \draw let \p1 = (current page.west), \p2 = (current page.east) in
      node[minimum width=\x2-\x1, minimum height=2cm, fill=none, anchor=north west, align=right, text width=\x2-\x1] at ($(current page.north west)$) {\large\bfseries\color{white} #2 \quad \quad \quad};       
  \end{tikzpicture}
}
